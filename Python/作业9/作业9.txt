import jieba

txt='''我们都是孤独的刺猬。只有频率相同的人，\
才能看见彼此内心深处不为人知的优雅。\
相信这世上一定有一个能感受到自己的人，\
那人未必是恋人，他可能是任何人。在诺大的世界里，\
我们会因为这份珍贵的懂得而不再孤独。

                                 ――《刺猬的优雅》'''

excludes=["我们","只有","频率"]

#数据整理
for i in "，。《》――\n":  #
    txt=txt.replace(i,'')

#加词
jieba.add_word("孤独的刺猬")


#分词
ls=jieba.lcut(txt)


#计数

d={}
for i in ls:
    if len(i)>=2:
        d[i]=d.get(i,0)+1

#去词

for word in excludes:
    del d[word]


#排序（按次数降序）

items = list(d.items())
items.sort(key=lambda x:x[1], reverse=True) 

#输出数据

for i,j in items[0:5]:
    print("{:<10}{:>5}".format(i,j))
